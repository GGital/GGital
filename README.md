        int Q; scanf("%d",&Q);
        for(int j = 0; j < (Q / 2) + (Q / 4) + 1; j++) {

                    if(k >= (Q/2) - (Q/4) - j && k <= (Q/2) - (Q/4) + j) printf("#");

                    else printf(".");

                        if(k >= (Q/2) - (Q/4) - j -1 && k <= (Q/2) - (Q/4) + j) printf("#");

                        else printf(".");
                    }
                }
                else if(j == Q/4) {
                    printf("#");
                }
                else if(j > Q/4) {
                    if(k >= j - Q/4 && k < Q - (j - Q/4)) {

        }
    }

}
